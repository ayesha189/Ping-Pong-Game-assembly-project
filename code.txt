			push di

			mov bx, 10       
			mov cx, 0        
 
			nextdigit:	mov dx, 0    
						div bx       
						add dl, 0x30 
						push dx      
						inc cx       
						cmp ax, 0    
						jnz nextdigit 
						
						mov ax, 0xb800

						mov di, [bp+6]

			pop cx 
			pop bx 
			pop ax 
			pop es

			;push dx
			
			mov ah, 1
			int 0x16
			jz exitInputP1

			mov ah, 0
			int 0x16

			je movePlayer1Up
